\section*{Problem 1: Definitions}

\begin{enumerate}[(a)]
	\item \textbf{Sufficiency principle:} The sufficient principle tell that a parameter is enough
	      to explain the population with the data.
	\item \textbf{Factorization theorem:} It is a tetanic where you separate the pdf in a way such that
	      you have the product of 2 function for all sample points $x$ and all parameter points $\theta$
	      $$
		      f(x|\theta) = g(T(x)|\theta)h(x).
	      $$
	\item \textbf{Likelihood Function:} It is the probability function that tells how well a parameter preforms given a dataset
	\item \textbf{Minimal sufficient:} it is the minimum data required to explain the parameter but still retaining all of the information
	\item \textbf{Complete statistics:} It is a complete information of the estimator without depending in outside factors.
\end{enumerate}

\[
	L(\theta; x_1, \dots, x_n) = \prod_{i=1}^n f_{X_i}(x_i \mid \theta)

	      $$f(x|\theta) =\frac{e^{-(x-\theta)}}{(1+e^{-(x-\theta)})^2}, \quad -\infty < x < \infty; - \infty<\theta<\infty$$
	\item \textbf{Cauchy:}
	      $$f(x|\theta) = \frac{1}{\pi} \frac{1}{1 + (x-\theta)^2}; \quad - \infty<x<\infty; \infty<\theta<\infty$$
	\item \textbf{Laplace (double exponential):}
	      $$f(x|\theta) = \frac{1}{2}e^{-|x-\theta|}, \quad - \infty<x<\infty; - \infty<\theta<\infty$$
\end{enumerate}
